    \begin{harj}
        Kirjoita seuraava:
        \begin{sample}
            Jos \(F\) on \(\sigma\)-algebra ja \(P\colon F\to\R\) todennäköisyys, niin tapahtumille \(A_1,A_2,\dotsc\in F\) pätee
            \[
                P\left(\bigcup_{i=1}^\infty A_i\right) \leq \sum_{i=1}^\infty P(A_i).
            \]
        \end{sample}
    \end{harj}
