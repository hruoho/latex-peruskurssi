        \[
            \verb-amsmath-,\quad \verb-amsthm-\quad \text{ja}\quad \verb-amssymb-

        \begin{verbatim}
        \usepackage{amsmath}
        \usepackage{amsthm}
        \usepackage{amssymb}
        \end{verbatim}
