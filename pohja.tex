% Dokumenttiluokka. Voi olla esim.

%  - report             (pidempiä tekstejä, kuten tutkielmia, varten)
% Lisäargumentteja:
%  - a4paper            (paperin koko)
%  - 10pt, 11pt, 12pt   (fontin koko)
%  - draft              (esim. näyttää ylipitkät rivit (overfull
%                        hbox-varoitukset); älä käytä lopullisessa työssä!)

%  - finnish            (ei dokumenttiluokan argumentti, mutta menee tätä
%                        kautta esim. babelille)

\usepackage{amsmath,amsthm,amssymb}
\usepackage{mathtools}

% Listojen kustomointi. Numeroidun listan numeroinnin voi vaihtaa
% aloittamalla listan komennolla
%   \begin{enumerate}[label=...]

% esim. komentoja \arabic* (tavalliset numerot), \alph* (pienaakkoset)

% b) ...
% jne.

\usepackage{lmodern}            % paranneltu fontti
%\usepackage{newpxtext,newpxmath} % erilainen fontti

% Lauseympäristöjen tyyli. Voi olla plain (oletus), definition tai remark.
\theoremstyle{plain}
\newtheorem{lause}{Lause}[section]      % numerointi alisteinen osiolle
\newtheorem{lemma}[lause]{Lemma}        % 'lause'-ympäristön kanssa jaettu numerointi
\newtheorem{prop}[lause]{Propositio}
\newtheorem{kor}[lause]{Korollaari}

\theoremstyle{definition}
\newtheorem{maar}[lause]{Määritelmä}
\newtheorem{esim}[lause]{Esimerkki}

\theoremstyle{remark}
\newtheorem{huom}[lause]{Huomautus}
